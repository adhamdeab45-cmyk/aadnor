rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    function signedIn() { return request.auth != null; }
    function isAdmin() {
      return signedIn() && (request.auth.token.email == 'adhamdeab45@gmail.com' || request.auth.token.email == 'adhamdeab2@gmail.com');
    }
    match /deposit-proofs/{uid}/{fileName} {
      allow create: if signedIn() && request.auth.uid == uid && request.resource.size < 8 * 1024 * 1024 && (request.resource.contentType.matches('image/.*') || request.resource.contentType == 'application/pdf');
      allow read: if signedIn() && (request.auth.uid == uid || isAdmin());
      allow update, delete: if false;
    }
    match /{allPaths=**} { allow read, write: if false; }
  }
}
